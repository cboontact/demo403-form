1. ภาพรวม — <form> คืออะไร

HTML Form คือส่วนที่ใช้รับข้อมูลจากผู้ใช้ — เช่น กรอกชื่อ เลือกเพศ อัปโหลดไฟล์ แล้วส่งข้อมูลไปยัง server หรือ Google Apps Script

<!-- โครงสร้างหลักของฟอร์ม -->
<form action="URL_ปลายทาง" method="post">

  <!-- ส่วนกรอกข้อมูล -->
  <label for="name">ชื่อ</label>
  <input type="text" id="name" name="name">

  <!-- ปุ่มส่ง -->
  <input type="submit" value="ส่งข้อมูล">

</form>
Attributeค่าหมายความว่า
action="..."URLปลายทางที่ข้อมูลจะถูกส่งไป เช่น Google Apps Script URL
method="post"post / getpost = ส่งข้อมูลซ่อนใน body (ปลอดภัยกว่า) | get = ต่อท้าย URL
for="..."ต้องตรงกับ id ของ inputเชื่อม <label> กับ <input> — คลิก label แล้วกล่องกรอกจะโฟกัสอัตโนมัติ
id="..."ชื่อไม่ซ้ำในหน้าตัวระบุ element สำหรับ CSS และ label
name="..."ชื่อตัวแปรชื่อที่ใช้รับข้อมูลฝั่ง server — ต้องมีทุกครั้ง

3. text / email / password / tel / url
<!-- ข้อความทั่วไป -->
<label for="fname">ชื่อ-นามสกุล</label>
<input type="text" id="fname" name="fname" placeholder="เช่น นายสมชาย ใจดี" required>

<!-- อีเมล -->
<label for="email">อีเมล</label>
<input type="email" id="email" name="email" placeholder="example@gmail.com">

<!-- รหัสผ่าน -->
<label for="pwd">รหัสผ่าน</label>
<input type="password" id="pwd" name="pwd" minlength="8">

<!-- เบอร์โทร -->
<label for="tel">เบอร์โทรศัพท์</label>
<input type="tel" id="tel" name="tel" pattern="[0-9]{10}" placeholder="0812345678">

<!-- URL -->
<label for="website">เว็บไซต์</label>
<input type="url" id="website" name="website" placeholder="https://example.com">
Attributeหน้าที่
placeholder="..."ข้อความตัวอย่างสีเทาที่แสดงเมื่อยังไม่กรอก
requiredบังคับกรอก — ส่งฟอร์มไม่ได้ถ้าช่องนี้ว่าง
minlength="8"จำนวนตัวอักษรขั้นต่ำ
pattern="[0-9]{10}"กำหนดรูปแบบด้วย Regex — ที่นี่คือตัวเลข 10 หลัก
4. number / range / date / time / datetime-local
<!-- ตัวเลข -->
<label for="age">อายุ</label>
<input type="number" id="age" name="age" min="1" max="120" value="18">

<!-- แถบเลื่อน -->
<label for="score">ความพึงพอใจ (1–10)</label>
<input type="range" id="score" name="score" min="1" max="10" step="1">

<!-- วันที่ -->
<label for="bday">วันเกิด</label>
<input type="date" id="bday" name="bday">

<!-- เวลา -->
<label for="appt">เวลานัด</label>
<input type="time" id="appt" name="appt">

<!-- วันที่ + เวลา -->
<label for="dt">วันและเวลา</label>
<input type="datetime-local" id="dt" name="dt">
Attributeหน้าที่
min="1" max="120"กำหนดค่าต่ำสุด-สูงสุด สำหรับ number และ range
step="1"กำหนดขั้นที่เลื่อนทีละเท่าไร — range ค่าเริ่มต้นคือ 1
value="18"ค่าเริ่มต้นที่แสดงในช่อง
5. radio & checkbox
<!-- radio: เลือกได้ 1 ตัว (name ต้องเหมือนกันทุกตัว) -->
<p>เพศ</p>
<input type="radio" id="male" name="gender" value="male">
<label for="male">ชาย</label>

<input type="radio" id="female" name="gender" value="female">
<label for="female">หญิง</label>

<input type="radio" id="other" name="gender" value="other">
<label for="other">ไม่ระบุ</label>

<!-- checkbox: เลือกได้หลายตัว (name ต่างกันได้) -->
<p>ความสนใจ</p>
<input type="checkbox" id="coding" name="interest" value="coding">
<label for="coding">เขียนโค้ด</label>

<input type="checkbox" id="design" name="interest" value="design" checked>
<label for="design">ออกแบบ</label>
ประเด็นradiocheckbox
เลือกได้กี่ตัว1 ตัวจากกลุ่มไม่จำกัด
nameต้องเหมือนกันทุกตัวในกลุ่มเหมือนหรือต่างก็ได้
valueค่าที่ส่งเมื่อเลือกค่าที่ส่งเมื่อติ๊ก
checkedค่าเริ่มต้นที่เลือกไว้ค่าเริ่มต้นที่ติ๊กไว้
กฎสำคัญ radio: ทุกตัวในกลุ่มเดียวกันต้องมี name เหมือนกัน — browser จะจัดให้เลือกได้ทีละตัวอัตโนมัติ
6. <select> & <textarea>
<!-- Dropdown เลือกจากรายการ -->
<label for="grade">ระดับชั้น</label>
<select id="grade" name="grade">
  <option value="">-- เลือกระดับชั้น --</option>
  <option value="m4">ม.4</option>
  <option value="m5" selected>ม.5</option>
  <option value="m6">ม.6</option>
</select>

<!-- Dropdown เลือกได้หลายตัว -->
<select name="subjects" multiple size="3">
  <option value="web">พัฒนาเว็บ</option>
  <option value="ai">Generative AI</option>
  <option value="data">Data Science</option>
</select>

<!-- กล่องข้อความหลายบรรทัด -->
<label for="msg">ข้อความ</label>
<textarea id="msg" name="msg" rows="4" cols="40" placeholder="พิมพ์ข้อความที่นี่..."></textarea>
Attributeใช้กับหน้าที่
<option value="m4">selectตัวเลือกแต่ละรายการ — value คือค่าที่ส่ง, ข้อความในแท็กคือที่แสดง
selectedoptionตัวเลือกที่แสดงค่าเริ่มต้น
multipleselectเลือกได้หลายตัว (กด Ctrl/Cmd ค้างไว้)
size="3"selectแสดงกี่แถวพร้อมกัน
rows="4"textareaความสูงกล่องข้อความ (จำนวนบรรทัด)
cols="40"textareaความกว้างกล่องข้อความ
7. file / color / hidden
<!-- อัปโหลดไฟล์ -->
<label for="photo">รูปโปรไฟล์</label>
<input type="file" id="photo" name="photo" accept="image/*">

<!-- อัปโหลดหลายไฟล์ -->
<input type="file" name="docs" accept=".pdf,.docx" multiple>

<!-- เลือกสี -->
<label for="favcolor">สีโปรด</label>
<input type="color" id="favcolor" name="favcolor" value="#16a34a">

<!-- hidden — ส่งค่าที่ซ่อนไว้กับฟอร์ม -->
<input type="hidden" name="userId" value="12345">
Attributeหน้าที่
accept="image/*"จำกัดประเภทไฟล์ที่เลือกได้ — image/* คือทุกนามสกุลรูป
accept=".pdf,.docx"จำกัดเฉพาะ PDF และ Word
multipleเลือกไฟล์ได้หลายไฟล์พร้อมกัน
type="hidden"ซ่อนจากผู้ใช้ — ใช้ส่งค่าคงที่เช่น userId หรือ timestamp
type="file" ต้องเพิ่ม enctype="multipart/form-data" ใน <form> ด้วย ไม่งั้นไฟล์จะไม่ถูกส่ง
8. ปุ่ม — submit / reset / button
<!-- ปุ่มส่งฟอร์ม (input) -->
<input type="submit" value="ส่งข้อมูล">

<!-- ปุ่มส่งฟอร์ม (button) — ใส่ไอคอนได้ -->
<button type="submit"> ส่งข้อมูล</button>

<!-- ปุ่มล้างฟอร์ม -->
<button type="reset"> ล้างข้อมูล</button>

<!-- ปุ่มทั่วไป (ไม่ส่งฟอร์ม — ใช้กับ JavaScript) -->
<button type="button" onclick="alert('คลิกแล้ว!')">ทดสอบ</button>
Typeพฤติกรรม
type="submit"ส่งฟอร์ม — ตรวจ validation ทุกช่องก่อนส่ง
type="reset"ล้างค่าทุกช่องกลับเป็นค่าเริ่มต้น
type="button"ปุ่มทั่วไป ไม่มีพฤติกรรมอัตโนมัติ — ใช้กับ JavaScript
9. Validation — บังคับกรอกให้ถูกต้อง
<!-- บังคับกรอก -->
<input type="text" name="name" required>

<!-- จำนวนตัวอักษร -->
<input type="text" name="nick" minlength="2" maxlength="20">

<!-- ช่วงตัวเลข -->
<input type="number" name="score" min="0" max="100">

<!-- รูปแบบด้วย Regex -->
<input type="text" name="zip" pattern="[0-9]{5}" title="รหัสไปรษณีย์ 5 หลัก">

<!-- ปิด validation (ส่งได้เลยโดยไม่ตรวจ) -->
<form novalidate> ... </form>
Attributeหน้าที่
requiredห้ามว่าง — browser แสดง popup เตือนอัตโนมัติ
minlength / maxlengthจำนวนตัวอักษรขั้นต่ำ/สูงสุด
min / maxค่าตัวเลขขั้นต่ำ/สูงสุด
pattern="[0-9]{5}"Regex — ตัวเลข 5 หลัก
title="..."ข้อความอธิบายใน popup เตือนเมื่อกรอกผิด pattern
novalidateปิดการตรวจ validation ทั้งหมดในฟอร์มนั้น
